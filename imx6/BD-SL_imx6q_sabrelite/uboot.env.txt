# generic params
bootdelay=3
stderr=serial,lcd
stdin=serial,usbkbd
stdout=serial,lcd

# CPU config
cpu=armv7
smp=on

# Console config
baudrate=115200
sttyconsole=ttyS0
ttyconsole=tty0

# Kernel/dtb filenames & load addresses
boot_it=bootm ${optee_addr_r} - ${fdt_addr_r}
fdt_addr_r=0x18000000
kernel_addr_r=0x12000000

# OP-TEE load address
optee_addr_r=0x20000000

# NFS/TFTP boot configuraton
# gatewayip=192.168.1.1
# netmask=255.255.255.0
# nfsserverip=192.168.1.100
# nfspath=/srv/nfs/rpi

# bootcmd & bootargs configuration
bootcmd=run mmcboot
load_kernel=load mmc 0:1 ${kernel_addr_r} zImage
load_dtb=load mmc 0:1 ${fdt_addr_r} imx6q-sabrelite.dtb
load_tee=load mmc 0:1 ${optee_addr_r} uTee
mmcboot=run set_bootargs; run load_kernel; run load_dtb; run load_tee; run boot_it
set_bootargs=setenv bootargs console=ttymxc1,${baudrate} root=/dev/mmcblk0p2 rootwait rw fixrtc
